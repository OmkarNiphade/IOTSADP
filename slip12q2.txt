// Q.2 Write a program to sense the available networks using Arduin
#include <WiFi.h>

void setup() {
  Serial.begin(115200);
  WiFi.mode(WIFI_STA);     // Station mode
  WiFi.disconnect();       // Disconnect previous WiFi
  delay(1000);

  Serial.println("Scanning WiFi networks...");
}

void loop() {
  int n = WiFi.scanNetworks();   // Scan nearby WiFi networks

  Serial.println("Scan Done!");
  if (n == 0) {
    Serial.println("No networks found");
  } else {
    for (int i = 0; i < n; i++) {
      Serial.print(i + 1);
      Serial.print(": ");
      Serial.print(WiFi.SSID(i));    // Network name
      Serial.print(" | Signal: ");
      Serial.print(WiFi.RSSI(i));    // Signal strength
      Serial.println(" dBm");
    }
  }

  Serial.println("-----------------------------");
  delay(5000);   // Scan again after 5 seconds
}
