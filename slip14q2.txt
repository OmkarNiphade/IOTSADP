int sensorPin = 2; 
int ledPin = 13; 
int sensorValue = 0;
void setup() {
 pinMode(sensorPin, INPUT); 
 pinMode(ledPin, OUTPUT); 
 Serial.begin(9600); 
}
void loop() {
 sensorValue = digitalRead(sensorPin); /
 if (sensorValue == HIGH) { 
 digitalWrite(ledPin, HIGH); 
 Serial.println("Vibration detected!");
 }
 else {
 digitalWrite(ledPin, LOW); 
 Serial.println("No vibration");
 }
 delay(200); 
}
