d) Area of rectangle, triangle, circle based on user input

shape = input()
if shape == "rectangle":
    l = float(input())
    w = float(input())
    print(l * w)

elif shape == "triangle":
    b = float(input())
    h = float(input())
    print(0.5 * b * h)

elif shape == "circle":
    r = float(input())
    print(3.14159 * r * r)
