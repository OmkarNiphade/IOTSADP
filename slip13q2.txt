char inputByte;
int led = 2;

void setup() {
  Serial.begin(9600);
  pinMode(led, OUTPUT);
}

void loop() {
  while (Serial.available() > 0) {
    inputByte = Serial.read();
    if (inputByte == '1') {
      digitalWrite(led, HIGH);
    } else if (inputByte == '2') {
      digitalWrite(led, LOW);
    }
  }
}
